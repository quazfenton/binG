async function readMd(filePath: string) {
  const raw = await fs.readFile(filePath,"utf8");
  const parsed = matter(raw);
  return {raw, parsed};
}

function makeSlug(title: string, relDir: string) {
  const base = slugify(title, {lower:true, strict:true});
  return path.join(relDir, base).replace(/\\/g,"/"); // unix style
}

function extractHeadings(raw: string) {
  const headings: {id:string;heading:string}[] = [];
  const lines = raw.split("\n");
  for (const line of lines) {
    const m = line.match(/^#{2,4}\s+(.*)/);
    if (m) {
      const heading = m[1].trim();
      const id = slugify(heading,{lower:true,strict:true});
      headings.push({id,heading});
    }
  }
  return headings;
}

function inferLayer(relPath: string) {
  if (relPath.includes("/onboarding/")) return "onboarding";
  if (relPath.includes("/examples/")) return "example";
  if (relPath.includes("/references/")) return "reference";
  return "core";
}

  }
  front.id = front.id || slug;
  front.title = title;

#!/usr/bin/env node
import fs from "fs-extra";
import path from "path";
import dotenv from "dotenv";
import crypto from "crypto";
import Fuse from "fuse.js";
import axios from "axios";
import { OpenAIApi, Configuration } from "openai";

dotenv.config();

const DOCS_ROOT = process.argv[2] || "docs";
const MANIFEST = path.join(DOCS_ROOT, "manifest.jsonl");
const OUT = path.join(DOCS_ROOT, "suggested-links.json");
const OPENAI_KEY = process.env.OPENAI_API_KEY;
if (!OPENAI_KEY) throw new Error("Set OPENAI_API_KEY in .env");

const conf = new Configuration({ apiKey: OPENAI_KEY });
const openai = new OpenAIApi(conf);

// helpers
function sha(s: string){ return crypto.createHash("sha1").update(s).digest("hex"); }
async function readManifest(): Promise<any[]> {
  const raw = await fs.readFile(MANIFEST, "utf8");
  return raw.trim().split("\n").map(l => JSON.parse(l));
}
async function getEmbedding(text: string): Promise<number[]> {
  const resp = await openai.createEmbedding({
    model: "text-embedding-3-small",
    input: text
  });
  return resp.data.data[0].embedding;
}
function cosSim(a:number[], b:number[]){
  let da=0, db=0, num=0;
  for(let i=0;i<a.length;i++){ num+=a[i]*b[i]; da+=a[i]*a[i]; db+=b[i]*b[i]; }
  return num/Math.sqrt(da*db+1e-12);
}

(async function main(){
  const items = await readManifest();
  // build lexical index (title + summary + tags)
  const fuse = new Fuse(items, { keys: ["title","summary","tags","path"], threshold: 0.3 });
  // compute or load embeddings
  const embDir = path.join(DOCS_ROOT,".embeddings");
  await fs.ensureDir(embDir);
  for (const it of items) {
    const embFile = path.join(embDir, sha(it.id)+".json");
    if (!await fs.pathExists(embFile)) {
      const text = `${it.title}\n\n${it.summary}\n\n${(it.tags||[]).join(" ")}`;
      const emb = await getEmbedding(text);
      await fs.writeJSON(embFile, emb);
      it._embedding = emb;
      await new Promise(r=>setTimeout(r,100)); // rate spacing
    } else {
      it._embedding = await fs.readJSON(embFile);
    }
  }

  const suggestions: any[] = [];
  for (const src of items) {
    // lexical candidates
    const lex = fuse.search(src.summary || src.title).slice(0,8).map(r=>r.item).filter(i=>i.id!==src.id);
    // semantic candidates by cosine sim
    const sims = items
      .filter(i=>i.id!==src.id)
      .map(i=>({item:i,score: cosSim(src._embedding, i._embedding)}))
      .sort((a,b)=>b.score-a.score)
      .slice(0,8)
      .map(x=>x.item);
    const candidates = Array.from(new Map([...lex, ...sims].map(i=>[i.id,i])).values());
    // compute confidence: blend lexical presence (title mention) and embedding score
    const recs = candidates.map(c=>{
      const embScore = cosSim(src._embedding, c._embedding);
      const lexicalScore = (src.title && c.title && src.title.toLowerCase().includes(c.title.toLowerCase())) ? 0.9 : 0;
      const pathRel = (src.path && c.path && src.path.startsWith(path.dirname(src.path))) ? 0.2 : 0;
      const confidence = Math.min(1, embScore*0.8 + lexicalScore*0.4 + pathRel*0.3);
      return {from: src.id, to: c.id, title: c.title, path: c.path, confidence: Number(confidence.toFixed(3))};
    }).sort((a,b)=>b.confidence-a.confidence).slice(0,10);
    suggestions.push({source: src.id, suggestions: recs});
  }

  await fs.writeJSON(OUT, suggestions, {spaces:2});
  console.log("Wrote suggested links:", OUT);
})().catch(err=>{ console.error(err); process.exit(1); });

function heuristicsClassify(src:any, tgt:any, suggestion: Suggestion) {
  // features
  const srcPath = src.path || "";
  const tgtPath = tgt.path || "";
  const srcTitle = (src.title||"").toLowerCase();
  const tgtTitle = (tgt.title||"").toLowerCase();
  const conf = suggestion.confidence;

  // keyword rules
  const exampleKeywords = ["example","sample","demo","guide","howto","tutorial","playground"];
  const implKeywords = ["implement","implementation","api","sdk","client","server","adapter","integration","usage","setup"];
  const dependsKeywords = ["depends","dependency","requires","prereq","prerequisite","auth","oauth","db","backend"];
  const referenceKeywords = ["reference","glossary","schema","spec","specification"];

  const containsAny = (s:string, arr:string[]) => arr.some(k => s.includes(k));

  // path proximity boost
  const sameDir = path.dirname(srcPath) === path.dirname(tgtPath);

  // title similarity (levenshtein normalized)
  const lev = levenshtein.get(srcTitle, tgtTitle);
  const levNorm = Math.max(0, 1 - lev / Math.max(srcTitle.length, tgtTitle.length, 1));

  // basic heuristics ordering
  if (containsAny(tgtTitle, exampleKeywords) || containsAny(tgtPath, "/examples/") || tgtPath.includes("/demo/")) {
    return {type: "example-of", score: Math.min(1, conf * 0.95 + (sameDir?0.05:0) + levNorm*0.1)};
  }
  if (containsAny(tgtTitle, implKeywords) || tgtPath.includes("/modules/") || tgtPath.includes("/api/") || tgtPath.includes("/sdk/")) {
    return {type: "implements", score: Math.min(1, conf * 0.9 + levNorm*0.05 + (sameDir?0.02:0))};
  }
  if (containsAny(tgtTitle, dependsKeywords) || containsAny(tgtPath, "/prereq/") || tgtPath.includes("/deps/")) {
    return {type: "depends-on", score: Math.min(1, conf * 0.9 + levNorm*0.03)};
  }
  if (containsAny(tgtTitle, referenceKeywords) || tgtPath.includes("/references/") || tgtPath.includes("/ref/") || tgtPath.includes("/glossary")) {
    return {type: "reference", score: Math.min(1, conf * 0.9 + levNorm*0.02)};
  }

  // fallback: if title substring match or short title highly similar -> duplicate/alias
  if (levNorm > 0.85 || srcTitle.includes(tgtTitle) || tgtTitle.includes(srcTitle)) {
    return {type: "duplicate", score: Math.min(1, conf * 0.9 + levNorm*0.15)};
  }

  // otherwise prefer "related" with lower score
  return {type: "related", score: Math.min(1, conf * 0.8 + levNorm*0.02)};
}

// Match symbols mentioned in a doc's content and return matches with counts
function matchSymbolsInDoc(symbolIndex: Record<string, {file:string;snippet?:string}[]>, docContent: string) {

// Classification heuristics
function heuristicsClassify(src:any, tgt:any, suggestion: Suggestion) {
  const srcPath = src.path || "";
  const tgtPath = tgt.path || "";
  const srcTitle = (src.title||"").toLowerCase();
  const tgtTitle = (tgt.title||"").toLowerCase();
  const conf = suggestion.confidence;

  const exampleKeywords = ["example","sample","demo","guide","howto","tutorial","playground"];
  const implKeywords = ["implement","implementation","api","sdk","client","server","adapter","integration","usage","setup"];
  const dependsKeywords = ["depends","dependency","requires","prereq","prerequisite","auth","oauth","db","backend"];
  const referenceKeywords = ["reference","glossary","schema","spec","specification"];

  const containsAny = (s:string, arr:string[]) => arr.some(k => s.includes(k));
  const sameDir = path.dirname(srcPath) === path.dirname(tgtPath);
  const lev = levenshtein.get(srcTitle, tgtTitle);
  const levNorm = Math.max(0, 1 - lev / Math.max(srcTitle.length, tgtTitle.length, 1));

  if (containsAny(tgtTitle, exampleKeywords) || containsAny(tgtPath, "/examples/") || tgtPath.includes("/demo/")) {
    return {type: "example-of", score: Math.min(1, conf * 0.95 + (sameDir?0.05:0) + levNorm*0.1)};
  }
  if (containsAny(tgtTitle, implKeywords) || tgtPath.includes("/modules/") || tgtPath.includes("/api/") || tgtPath.includes("/sdk/")) {
    return {type: "implements", score: Math.min(1, conf * 0.9 + levNorm*0.05 + (sameDir?0.02:0))};
  }
  if (containsAny(tgtTitle, dependsKeywords) || containsAny(tgtPath, "/prereq/") || tgtPath.includes("/deps/")) {
    return {type: "depends-on", score: Math.min(1, conf * 0.9 + levNorm*0.03)};
  }
  if (containsAny(tgtTitle, referenceKeywords) || tgtPath.includes("/references/") || tgtPath.includes("/ref/") || tgtPath.includes("/glossary")) {
    return {type: "reference", score: Math.min(1, conf * 0.9 + levNorm*0.02)};
  }
  if (levNorm > 0.85 || srcTitle.includes(tgtTitle) || tgtTitle.includes(srcTitle)) {
    return {type: "duplicate", score: Math.min(1, conf * 0.9 + levNorm*0.15)};
  }
  return {type: "related", score: Math.min(1, conf * 0.8 + levNorm*0.02)};
}
